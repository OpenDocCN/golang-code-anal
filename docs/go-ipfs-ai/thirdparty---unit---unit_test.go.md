// 导入测试包
import "testing"

// 测试字节单位转换函数
func TestByteSizeUnit(t *testing.T) {
    // 检查 KB 单位是否等于 1024
    if 1*KB != 1*1024 {
        t.Fatal(1 * KB)
    }
    // 检查 MB 单位是否等于 1024*1024
    if 1*MB != 1*1024*1024 {
        t.Fail()
    }
    // 检查 GB 单位是否等于 1024*1024*1024
    if 1*GB != 1*1024*1024*1024 {
        t.Fail()
    }
    // 检查 TB 单位是否等于 1024*1024*1024*1024
    if 1*TB != 1*1024*1024*1024*1024 {
        t.Fail()
    }
    // 检查 PB 单位是否等于 1024*1024*1024*1024*1024
    if 1*PB != 1*1024*1024*1024*1024*1024 {
        t.Fail()
    }
    // 检查 EB 单位是否等于 1024*1024*1024*1024*1024*1024
    if 1*EB != 1*1024*1024*1024*1024*1024*1024 {
        t.Fail()
    }
}
